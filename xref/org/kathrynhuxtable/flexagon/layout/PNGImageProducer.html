127     public synchronized void addConsumer(ImageConsumer ic) {
128         if (theConsumers.contains(ic)) {
129             return;
130         }
131 
132         theConsumers.addElement(ic);
133 
134         try {
135             initConsumer(ic);
136             sendPixels(ic, 0, 0, width, height);
137             if ((complete) && (isConsumer(ic))) {
138                 if (error) {
139                     ic.imageComplete(ImageConsumer.IMAGEERROR);
140                 } else {
141                     ic.imageComplete(ImageConsumer.STATICIMAGEDONE);
142                 }
143                 removeConsumer(ic);
144             }
145         } catch (Exception e) {
146             if (isConsumer(ic)) {
147                 ic.imageComplete(ImageConsumer.IMAGEERROR);
148             }
149         }
150     }

152     private void blockFill(int rowStart) {
153         int counter;
154         int dw = dataWidth;
155         int pass = this.pass;
156         int w = blockWidth[pass];
157         int sCol = startingCol[pass];
158         int cInc = colInc[pass];
159         int wInc = cInc - w;
160         int maxW = rowStart + dw - w;
161         int len;
162 
163         int h = blockHeight[pass];
164         int maxH = rowStart + (dw * h);
165         int startPos = rowStart + sCol;
166 
167         counter = startPos;
168 
169         if (colorType == 3) {
170             byte bpix[] = bpixels;
171             byte pixel;
172             len = bpix.length;
173 
174             for (; counter <= maxW;) {
175                 int end = counter + w;
176                 pixel = bpix[counter++];
177                 for (; counter < end; counter++) {
178                     bpix[counter] = pixel;
179                 }
180                 counter += wInc;
181             }
182             maxW += w;
183 
184             if (counter < maxW) {
185                 for (pixel = bpix[counter++]; counter < maxW; counter++) {
186                     bpix[counter] = pixel;
187                 }
188             }
189 
190             if (len < maxH)
191                 maxH = len;
192 
193             for (counter = startPos + dw; counter < maxH; counter += dw) {
194                 System.arraycopy(bpix, startPos, bpix, counter, dw - sCol);
195             }
196         } else {
197             int ipix[] = ipixels;
198             int pixel;
199             len = ipix.length;
200 
201             for (; counter <= maxW;) {
202                 int end = counter + w;
203                 pixel = ipix[counter++];
204                 for (; counter < end; counter++) {
205                     ipix[counter] = pixel;
206                 }
207                 counter += wInc;
208             }
209 
210             maxW += w;
211 
212             if (counter < maxW) {
213                 for (pixel = ipix[counter++]; counter < maxW; counter++) {
214                     ipix[counter] = pixel;
215                 }
216             }
217 
218             if (len < maxH)
219                 maxH = len;
220 
221             for (counter = startPos + dw; counter < maxH; counter += dw) {
222                 System.arraycopy(ipix, startPos, ipix, counter, dw - sCol);
223             }
224         }
225     }

227     private boolean filterRow(byte inbuf[], int pix[], int upix[],
228                                 int rowFilter, int boff) {
229         int rowWidth = pix.length;
230 
231         switch (rowFilter) {
232           case 0:
233             for (int x = 0; x < rowWidth; x++) {
234                 pix[x] = 0xff & inbuf[x];
235             }
236             break;
237           case 1: {
238             int x = 0;
239             for ( ; x < boff; x++) {
240                 pix[x] = 0xff & inbuf[x];
241             }
242             for ( ; x < rowWidth; x++) {
243                 pix[x] = 0xff & (inbuf[x] + pix[x - boff]);
244             }
245             break;
246           }
247           case 2: {
248             if (upix != null){
249                 for (int x = 0; x < rowWidth; x++) {
250                     pix[x] = 0xff & (upix[x] + inbuf[x]);
251                 }
252             } else {
253                 for (int x = 0; x < rowWidth; x++) {
254                     pix[x] = 0xff & inbuf[x];
255                 }
256             }
257             break;
258           }
259           case 3: {
260             if (upix != null) {
261                 int x = 0;
262                 for ( ; x < boff; x++) {
263                     int rval = upix[x];
264                     pix[x] = 0xff & ((rval>>1) + inbuf[x]);
265                 }
266                 for ( ; x < rowWidth; x++) {
267                     int rval = upix[x] + pix[x - boff];
268                     pix[x] = 0xff & ((rval>>1) + inbuf[x]);
269                 }
270             } else {
271                 int x = 0;
272                 for ( ; x < boff; x++) {
273                     pix[x] = 0xff & inbuf[x];
274                 }
275                 for ( ; x < rowWidth; x++) {
276                     int rval = pix[x - boff];
277                     pix[x] = 0xff & ((rval>>1) + inbuf[x]);
278                 }
279             }
280             break;
281           }
282           case 4: {
283             if (upix != null) {
284                 int x = 0;
285                 for ( ; x < boff; x++) {
286                     pix[x] = 0xff & (upix[x] + inbuf[x]);
287                 }
288                 for ( ; x < rowWidth; x++) {
289                     int a, b, c, p, pa, pb, pc, rval;
290                     a = pix[x - boff];
291                     b = upix[x];
292                     c = upix[x - boff];
293                     p = a + b - c;
294                     pa = p > a ? p - a : a - p;
295                     pb = p > b ? p - b : b - p;
296                     pc = p > c ? p - c : c - p;
297                     if ((pa <= pb) && (pa <= pc)) {
298                         rval = a;
299                     } else if (pb <= pc) {
300                         rval = b;
301                     } else {
302                         rval = c;
303                     }
304                     pix[x] = 0xff & (rval + inbuf[x]);
305                 }
306             } else {
307                 int x = 0;
308                 for ( ; x < boff; x++) {
309                     pix[x] = 0xff & inbuf[x];
310                 }
311                 for ( ; x < rowWidth; x++) {
312                     int rval = pix[x - boff];
313                     pix[x] = 0xff & (rval + inbuf[x]);
314                 }
315             }
316             break;
317           }
318           default:
319             return false;
320         }
321         return true;
322     }

332     private void handleChunk() throws IOException {
333         if (needChunkInfo) {
334             chunkLength = inputStream.readInt();
335             chunkType = inputStream.readInt();
336             needChunkInfo = false;
337         }
338 
339         // Guarantee that chunks can't overread their bounds
340         /*inputStream = new DataInputStream(
341                         new MeteredInputStream(underlyingStream,
342                                                 chunkLength));
343         */
344         switch (chunkType) {
345           case CHUNK_bKGD:
346             handlebKGD();
347             break;
348           case CHUNK_cHRM:
349             handlecHRM();
350             break;
351           case CHUNK_gAMA:
352             handlegAMA();
353             break;
354           case CHUNK_hIST:
355             handlehIST();
356             break;
357           case CHUNK_IDAT:
358             handleIDAT();
359             break;
360           case CHUNK_IEND:
361             handleIEND();
362             break;
363           case CHUNK_IHDR:
364             handleIHDR();
365             break;
366           case CHUNK_pHYs:
367             handlepHYs();
368             break;
369           case CHUNK_PLTE:
370             handlePLTE();
371             break;
372           case CHUNK_sBIT:
373             handlesBIT();
374             break;
375           case CHUNK_tEXt:
376             handletEXt();
377             break;
378           case CHUNK_tIME:
379             handletIME();
380             break;
381           case CHUNK_tRNS:
382             handletRNS();
383             break;
384           case CHUNK_zTXt:
385             handlezTXt();
386             break;
387           default:
388             System.err.println("unrecognized chunk type " +
389                                 Integer.toHexString(chunkType) + ". skipping");
390             inputStream.skip(chunkLength);
391         }
392         //inputStream = new DataInputStream(underlyingStream);
393         int crc = inputStream.readInt();
394         needChunkInfo = true;
395     }

405     /**
406      * Handle the image data chunks.  Note that sending info to ImageConsumers
407      * is delayed until the first IDAT chunk is seen.  This allows for any
408      * changes in ancilliary chunks that may alter the overall properties of
409      * the image, such as aspect ratio altering the image dimensions.  We
410      * assume that all ancilliary chunks which have these effects will appear
411      * before the first IDAT chunk, and once seen, image properties are set in
412      * stone.
413      **/
414 
415     private void handleIDAT() throws IOException {
416         if (!infoAvailable) {
417             if (width == -1)
418                 width = dataWidth;
419             if (height == -1)
420                 height = dataHeight;
421             setColorModel();
422             if (interlaceMethod != 0)
423                 multipass = true;
424 
425             Vector consumers;
426             synchronized (this) {
427                 infoAvailable = true;
428                 consumers = (Vector) theConsumers.clone();
429             }
430             for (int i = 0; i < consumers.size(); i++) {
431                 initConsumer((ImageConsumer) consumers.elementAt(i));
432             }
433         }
434 
435         readImageData();
436 
437         sendPixels(0, 0, width, height);
438     }

493     private void handletRNS() throws IOException {
494         int chunkLen = chunkLength;
495 		if (palette == null)
496 			throw new IOException("tRNS chunk encountered before pLTE");
497 
498 		int len = palette.length;
499 
500         switch (colorType) {
501           case 3: {
502             transparency = true;
503             int transLength = len/3;
504             byte[] trans = new byte[transLength];
505             inputStream.readFully(trans, 0, chunkLength);
506 
507             byte b = (byte) 0xff;
508 
509             for (int i = len; i < transLength; i++) {
510                 trans[i] = b;
511             }
512 
513             byte[] newPalette = new byte[len + transLength];
514 
515             for (int i = newPalette.length; i > 0;) {
516                 newPalette[--i] = trans[--transLength];
517                 newPalette[--i] = palette[--len];
518                 newPalette[--i] = palette[--len];
519                 newPalette[--i] = palette[--len];
520             }
521 
522             palette = newPalette;
523             break;
524           }
525           default:
526             inputStream.skip(chunkLength);
527         }
528     }

570     private void insertGreyPixels(int pix[], int offset, int samples) {
571         int p = pix[0];
572         int ipix[] = ipixels;
573         int cInc = colInc[pass];
574         int rs = 0;
575 
576         switch (colorType) {
577           case 0: {
578             switch (depth) {
579               case 1: {
580                 for (int j = 0; j < samples; j++, offset += cInc) {
581                     if (rs != 0) {
582                         rs--;
583                     } else {
584                         rs = 7;
585                         p = pix[j>>3];
586                     }
587                     ipix[offset] = (p>>rs) & 0x1;
588                 }
589                 break;
590               }
591               case 2: {
592                 for (int j = 0; j < samples; j++, offset += cInc) {
593                     if (rs != 0) {
594                         rs -= 2;
595                     } else {
596                         rs = 6;
597                         p = pix[j>>2];
598                     }
599                     ipix[offset] = (p>>rs) & 0x3;
600                 }
601                 break;
602               }
603               case 4: {
604                 for (int j = 0; j < samples; j++, offset += cInc) {
605                     if (rs != 0) {
606                         rs = 0;
607                     } else {
608                         rs = 4;
609                         p = pix[j>>1];
610                     }
611                     ipix[offset] = (p>>rs) & 0xf;
612                 }
613                 break;
614               }
615               case 8: {
616                 for (int j = 0; j < samples; offset += cInc) {
617                     ipix[offset] = (byte) pix[j++];
618                 }
619                 break;
620               }
621               case 16: {
622                 samples = samples<<1;
623                 for (int j = 0; j < samples; j += 2, offset += cInc) {
624                     ipix[offset] = pix[j];
625                 }
626                 break;
627               }
628               default:
629                 break;
630             }
631             break;
632           }
633           case 4: {
634             if (depth == 8) {
635                 for (int j = 0; j < samples; offset += cInc) {
636                     ipix[offset] = (pix[j++]<<8) | pix[j++];
637                 }
638             } else {
639                 samples = samples<<1;
640                 for (int j = 0; j < samples; j += 2, offset += cInc) {
641                     ipix[offset] = (pix[j]<<8) | pix[j+=2];
642                 }
643             }
644             break;
645           }
646         }
647     }

703     private void insertPixels(int pix[], int offset, int samples) {
704         switch (colorType) {
705           case 0:
706           case 4: {
707             insertGreyPixels(pix, offset, samples);
708             break;
709           }
710           case 2: {
711             int j = 0;
712             int ipix[] = ipixels;
713             int cInc = colInc[pass];
714             if (depth == 8) {
715                 for (j = 0; j < samples; offset += cInc) {
716                     ipix[offset] = (pix[j++]<<16) | (pix[j++]<<8) | pix[j++];
717                 }
718             } else {
719                 samples = samples<<1;
720                 for (j = 0; j < samples; j += 2, offset += cInc) {
721                     ipix[offset] = (pix[j]<<16) | (pix[j+=2]<<8) | pix[j+=2];
722                 }
723             }
724             break;
725           }
726           case 3: {
727             insertPalettedPixels(pix, offset, samples);
728             break;
729           }
730           case 6: {
731             int j = 0;
732             int ipix[] = ipixels;
733             int cInc = colInc[pass];
734             if (depth == 8) {
735                 for (j = 0; j < samples; offset += cInc) {
736                     ipix[offset] = (pix[j++]<<16) | (pix[j++]<<8) | pix[j++] |
737                                     (pix[j++]<<24);
738                 }
739             } else {
740                 samples = samples<<1;
741                 for (j = 0; j < samples; j += 2, offset += cInc) {
742                     ipix[offset] = (pix[j]<<16) | (pix[j+=2]<<8) | pix[j+=2] |
743                                     (pix[j+=2]<<24);
744                 }
745             }
746             break;
747           }
748           default:
749             break;
750         }
751     }

766      **/

768     private void readImageData() throws IOException {
769         long time = System.currentTimeMillis();
770 
771         InputStream dataStream = new SequenceInputStream(
772                                         new IDATEnumeration(this));
773 
774         DataInputStream dis = new DataInputStream(
775                                 new BufferedInputStream(
776                                     new InflaterInputStream(dataStream,
777                                                             new Inflater())));
778 
779         int bps, filterOffset;
780 
781         switch (colorType) {
782           case 0:
783           case 3:
784             bps = depth;
785             break;
786           case 2:
787             bps = 3 * depth;
788             break;
789           case 4:
790             bps = depth<<1;
791             break;
792           case 6:
793             bps = depth<<2;
794             break;
795           default:
796             // should never happen
797             throw new IOException("Unknown color type encountered.");
798         }
799 
800         filterOffset = (bps + 7)>>3;
801 
802         for (pass = (multipass ? 1 : 0); pass < 8; pass++) {
803             int pass = this.pass;
804             int rInc = rowInc[pass];
805             int cInc = colInc[pass];
806             int sCol = startingCol[pass];
807 
808             int val = ((dataWidth - sCol + cInc - 1) / cInc);
809 
810             int samples = val * filterOffset;
811             int rowSize = (val * bps)>>3;
812 
813             int sRow = startingRow[pass];
814 
815             if ((dataHeight <= sRow) || (rowSize == 0))
816                 continue;
817 
818             int sInc = rInc * dataWidth;
819 
820             byte inbuf[] = new byte[rowSize];
821             int pix[] = new int[rowSize];
822             int upix[] = null;
823             int temp[] = new int[rowSize];
824 
825             // next Y value and number of rows to report to sendPixels
826             int nextY = sRow;
827             int rows = 0;
828             int rowStart = sRow * dataWidth;
829 
830             for (int y = sRow; y < dataHeight; y += rInc, rowStart += sInc) {
831                 rows += rInc;
832 
833                 int rowFilter = dis.read();
834                 dis.readFully(inbuf);
835 
836                 if (!filterRow(inbuf, pix, upix, rowFilter, filterOffset)) {
837                     throw new IOException("Unknown filter type: " + rowFilter);
838                 }
839 
840                 insertPixels(pix, rowStart + sCol, samples);
841 
842                 if (multipass && (pass < 6)) {
843                     blockFill(rowStart);
844                 }
845 
846                 upix = pix;
847                 pix = temp;
848                 temp = upix;
849 
850                 if (!completePasses) {
851                     long newTime = System.currentTimeMillis();
852                     if ((newTime - time) > updateDelay) {
853                         sendPixels(0, nextY, width, rows);
854                         rows = 0;
855                         nextY = y + rInc;
856                         time = newTime;
857                     }
858                 }
859                 Thread.yield();
860             }
861             if (!multipass)
862                 break;
863             if (completePasses || (rows > 0)) {
864                 sendPixels(0, 0, width, height);
865                 time = System.currentTimeMillis();
866             }
867         }
868         while(dis.read() != -1)
869             System.err.println("Leftover data encountered.");
870     }

947     private void setColorModel() throws IOException {
948         int mask = 0;
949         switch (depth) {
950           case 1:
951             mask = 0x1;
952             break;
953           case 2:
954             mask = 0x3;
955             break;
956           case 4:
957             mask = 0xf;
958             break;
959           case 8:
960           case 16:
961             mask = 0xff;
962             break;
963         }
964 
965         int count = width * height;
966 
967         switch (colorType) {
968           case 3:
969             if (palette == null)
970                 throw new IOException("No palette located");
971             bpixels = new byte[count];
972             pixels = bpixels;
973             if (transparency) {
974                 model = new IndexColorModel(depth, palette.length/4, palette,
975                                             0, true);
976             } else {
977                 model = new IndexColorModel(depth, palette.length/3, palette,
978                                             0, false);
979             }
980             break;
981           case 0:
982             ipixels = new int[count];
983             pixels = ipixels;
984             if (depth < 8) {
985                 model = new DirectColorModel(depth, mask, mask, mask);
986             } else {
987                 model = new DirectColorModel(8, mask, mask, mask);
988             }
989             break;
990           case 2:
991             ipixels = new int[count];
992             pixels = ipixels;
993             model = new DirectColorModel(24, 0xff0000, 0xff00, 0xff);
994             break;
995           case 4:
996             int smask;
997             if (depth < 8) {
998                 smask = mask << depth;
999                 model = new DirectColorModel(depth * 2, smask, smask, smask,
1000                                                 mask);
1001             } else {
1002                 smask = mask << 8;
1003                 model = new DirectColorModel(16, smask, smask, smask, mask);
1004             }
1005             ipixels = new int[count];
1006             pixels = ipixels;
1007             break;
1008           case 6:
1009             ipixels = new int[count];
1010             pixels = ipixels;
1011             model = ColorModel.getRGBdefault();
1012             break;
1013           default:
1014             throw new IOException("Image has unknown color type");
1015         }
