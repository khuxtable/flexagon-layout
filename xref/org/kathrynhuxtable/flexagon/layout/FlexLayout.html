487 	private int drawRow(int y, int face, int facet, int rot) {
488 		if (y % 10 == 0)
489 			System.err.print("On row " + y + "       \r");
490 
491 		if (backFlag)
492 			face -= numFaces/2;
493 
494 		rot /= 60;
495 
496 		boolean upsideDown = (rot & 1) == 1;
497 
498 		if (upsideDown) {
499 			rot = (rot + 3) % 6;		// rotate 180 degrees
500 			y = height/2 - y;			// and flip
501 		}
502 
503 		int pixels[] = new int[width];
504 
505 		int w = faces[face].getRow(y, facet, rot, pixels);
506 
507 		if (upsideDown) {
508 			for (int i = w - 1; i >= 0; i--)
509 				drawPixel(pixels[i]);
510 		}
511 		else {
512 			for (int i = 0; i < w; i++)
513 				drawPixel(pixels[i]);
514 		}
515 
516 		return w;
517 	}

522 	private int drawStart(int y, boolean pointDown) {
523 		if (pointDown)
524 			y = height/2 - y;
525 
526 		int end = (int) Math.floor(y * width / 2.0 / height);
527 
528 		for (int dx = 0; dx < end; dx++) {
529 			drawPixel(0);
530 		}
531 
532 		return end;
533 	}

539 	private int drawFinish(int x, int y) {
540 		int end = (int) (2.75 * width);
541 
542 		for (int dx = x; dx < end; dx++) {
543 			drawPixel(0);
544 		}
545 
546 		return end - x;
547 	}

550 	private void drawPixel(int pixel) {
551 		int red   = (pixel >> 16) & 0xFF;
552 		int green = (pixel >>  8) & 0xFF;
553 		int blue  = (pixel >>  0) & 0xFF;
554 
555 		try {
556 			out.write(red);
557 			out.write(green);
558 			out.write(blue);
559 		}
560 		catch (IOException e) {
561 			System.err.println("IO exception writing to image file");
562 			System.exit(0);
563 		}
564 	}
