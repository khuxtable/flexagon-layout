150     public void generate() {
151         // Get the image pixels for each face
152 
153         faces = new FlexagonFace[numFaces / 2];
154 
155         backFlag = false;
156 
157         getImages();
158 
159         if (numFaces == 6) {
160             drawFrontHexa();
161         } else {
162             drawFrontLeftDodeca();
163             drawFrontRightDodeca();
164         }
165 
166         backFlag = true;
167 
168         getImages();
169 
170         if (numFaces == 6) {
171             drawBackHexa();
172         } else {
173             drawBackLeftDodeca();
174             drawBackRightDodeca();
175         }
176 
177     }
