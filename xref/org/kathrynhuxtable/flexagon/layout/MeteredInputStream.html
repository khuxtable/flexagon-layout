15  public class MeteredInputStream extends FilterInputStream {

19      public MeteredInputStream(InputStream in, int size) {
20          super(in);
21          bytesLeft = size;

24      /**
25       * Reads a byte. Will block if no input is available.
26       * @return 	the byte read, or -1 if the end of the
27       *		stream is reached.
28       * @exception IOException If an I/O error has occurred.
29       */
30  
31      public final int read() throws IOException {
32          if (bytesLeft > 0) {
33              int val = in.read();
34              if (val != -1)
35                  bytesLeft--;
36      	    return val;
37      	}
38      	return -1;
39      }

43       * Blocks until some input is available.

46       * 		when the end of the stream is reached.

54      /**
55       * Reads into an array of bytes.
56       * Blocks until some input is available.
57       * This method should be overridden in a subclass for
58       * efficiency (the default implementation reads 1 byte
59       * at a time).
60       * @param b	the buffer into which the data is read
61       * @param off the start offset of the data
62       * @param len the maximum number of bytes read
63       * @return  the actual number of bytes read. Returns -1
64       * 		when the end of the stream is reached.
65       * @exception IOException If an I/O error has occurred.
66       */
67  
68      public final int read(byte b[], int off, int len) throws IOException {
69          if (bytesLeft > 0) {
70              len = (len > bytesLeft ? bytesLeft : len);
71              int read = in.read(b, off, len);
72              if (read > 0)
73                  bytesLeft -= read;
74              return read;
75          }
76          return -1;
77      }

79      /**
80       * Skips bytes of input.
81       * @param n 	bytes to be skipped
82       * @return	actual number of bytes skipped
83       * @exception IOException If an I/O error has occurred.
84       */
85      public final long skip(long n) throws IOException {
86          n = (n > bytesLeft ? bytesLeft : n);
87          long skipped = in.skip(n);
88          if (skipped > 0)
89              bytesLeft -= skipped;
90          return skipped;
91      }

111     /**
112      * Marks the current position in the input stream.  A subsequent
113      * call to reset() will reposition the stream at the last
114      * marked position so that subsequent reads will re-read
115      * the same bytes.  The stream promises to allow readlimit bytes
116      * to be read before the mark position gets invalidated.
117      * @param readlimit the maximum limit of bytes allowed tobe read before the
118      * mark position becomes invalid.
119      */
120     public final void mark(int readlimit) {
121         marked = bytesLeft;
122 	    in.mark(readlimit);
123     }

125     /**
126      * Repositions the stream to the last marked position.  If the
127      * stream has not been marked, or if the mark has been invalidated,
128      * an IOException is thrown.  Stream marks are intended to be used in
129      * situations where you need to read ahead a little to see what's in
130      * the stream.  Often this is most easily done by invoking some
131      * general parser.  If the stream is of the type handled by the
132      * parse, it just chugs along happily.  If the stream is not of
133      * that type, the parser should toss an exception when it fails.
134      * If this happens within readlimit bytes,  it allows the outer
135      * code to reset the stream and try another parser.
136      */
137     public final void reset() throws IOException {
138         in.reset();
139         bytesLeft = marked;
140     }
