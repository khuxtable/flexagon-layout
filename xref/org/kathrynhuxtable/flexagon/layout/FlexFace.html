56  	public FlexFace(String imageName, int width, int height)
57  						throws NullPointerException {
58  		this.out = out;
59  		this.width = width;
60  		this.height = height;
61  
62  		for (int src = 0; src < 6; src++) {
63  			ctrX[src] = Math.round(width  * ((src % 3) + 1) / 4.0);
64  
65  			ctrY[src] = Math.round(height * ((src < 3 ? 0 : 0.5) +
66  											((src & 1) == 0 ? 2 : 1) / 6.0));
67  		}
68  
69  		getImagePixels(imageName);
70  	}

73  	private void getImagePixels(String imageName) {
74  		ImageProducer ip;
75  
76  		try {
77  	        ip = new PNGImageProducer(new FileInputStream(imageName));
78  		}
79  		catch (FileNotFoundException e) {
80  			throw new NullPointerException("couldn't get image " + imageName);
81  		}
82  
83  		if (ip == null) {
84  			throw new NullPointerException("couldn't get image " + imageName);
85  		}
86  
87  		raster = new int[width * height];
88  
89  		PixelGrabber pg = new PixelGrabber(ip, 0, 0, width, height, raster,
90  										   0, width);
91  
92  		try {
93  			pg.grabPixels();
94  		}
95  		catch (InterruptedException e) {
96  			raster = null;
97  			throw new NullPointerException("interrupted waiting for pixels from " + imageName);
98  		}
99  
100 		if ((pg.status() & ImageObserver.ABORT) != 0) {
101 			raster = null;
102 			throw new NullPointerException("image fetch aborted or returned error for " + imageName);
103 		}
104 	}

125 	private void rotate(int facet, int rot, double x, double y, double[] ret) {
126 		// right-handed coordinate system.
127 
128 		x -= ctrX[facet];
129 		y -= ctrY[facet];
130 
131 		ret[0] = cos[rot] * x - sin[rot] * y + ctrX[facet];
132 		ret[1] = cos[rot] * y + sin[rot] * x + ctrY[facet];
133 	}

136 	private int getPixel(int x, int y) {
137 		if (x < 0 || y < 0 || x >= width || y >= height)
138 			return 0;
139 
140 		return raster[y * width + x];
141 	}

144 	private int averagePixel(double weight, int p1, int p2) {
145 		int r1 = (p1 >> 16) & 0xFF;
146 		int g1 = (p1 >>  8) & 0xFF;
147 		int b1 = (p1      ) & 0xFF;
148 		int r2 = (p2 >> 16) & 0xFF;
149 		int g2 = (p2 >>  8) & 0xFF;
150 		int b2 = (p2      ) & 0xFF;
151 
152 		int r = (int) (weight * r1 + (1.0 - weight) * r2);
153 		int g = (int) (weight * g1 + (1.0 - weight) * g2);
154 		int b = (int) (weight * b1 + (1.0 - weight) * b2);
155 
156 		return (r << 16) | (g << 8) | b;
157 	}

160 	private void setFacetRow(int facet, int rot, int x, int new_x, int y,
161 							 int pixels[]) {
162 		if (x == new_x)
163 			return;
164 
165 		// Calculate starting point for row
166 
167 		rotate(facet, rot, x, y, coord);
168 
169 		double x1 = coord[0];
170 		double y1 = coord[1];
171 
172 		// Calculate ending point for row
173 
174 		rotate(facet, rot, new_x, y, coord);
175 
176 		double x2 = coord[0];
177 		double y2 = coord[1];
178 
179 		// Calculate increments for pixels in row
180 
181 		double xinc = (x2 - x1) / (new_x - x);
182 		double yinc = (y2 - y1) / (new_x - x);
183 
184 		// Calculate increments for perpendicular
185 
186 		rotate(facet, rot, x, y + 1, coord);
187 
188 		double x3 = coord[0];
189 		double y3 = coord[1];
190 
191 		double xperp = x3 - x1;
192 		double yperp = y3 - y1;
193 
194 		for (int i = 0; x < new_x; x++) {
195 			int sx = (int) Math.round(x1);
196 			int sy = (int) Math.round(y1);
197 
198 			int p1 = getPixel(sx, sy);
199 			int p2 = p1;
200 			double w;
201 
202 			if (Math.abs(x1 - sx) > Math.abs(y1 - sy)) {
203 				int sx2 = sx + (x1 - sx >= 0 ? 1 : -1);
204 				p2 = getPixel(sx2, sy);
205 				w = 1.0 - (double) Math.abs(x1 - sx);
206 			}
207 			else {
208 				int sy2 = sy + (y1 - sy >= 0 ? 1 : -1);
209 				p2 = getPixel(sx, sy2);
210 				w = 1.0 - (double) Math.abs(y1 - sy);
211 			}
212 
213 			pixels[i++] = averagePixel(w, p1, p2);
214 
215 			x1 += xinc;
216 			y1 += yinc;
217 		}
218 	}

225 	public int getRow(int y, int facet, int rot, int pixels[]) {
226 
227 		double beg_factor;
228 		double end_factor;
229 
230 		switch (facet) {
231 		case 0:
232 			beg_factor = 0.5;
233 			end_factor = 0.5;
234 			break;
235 		case 1:
236 			beg_factor = 0.5;
237 			end_factor = 1.5;
238 			break;
239 		case 2:
240 			beg_factor = 1.5;
241 			end_factor = 1.5;
242 			break;
243 		case 3:
244 			beg_factor = 0.0;
245 			end_factor = 1.0;
246 			break;
247 		case 4:
248 			beg_factor = 1.0;
249 			end_factor = 1.0;
250 			break;
251 		default: // case 5:
252 			beg_factor = 1.0;
253 			end_factor = 2.0;
254 			break;
255 		}
256 
257 		beg_factor *= height;
258 		end_factor *= height;
259 
260 		if ((facet & 1) == 0) {
261 			beg_factor -= y;
262 			end_factor += y;
263 		}
264 		else {
265 			beg_factor += y;
266 			end_factor -= y;
267 		}
268 
269 		int beg = (int) Math.floor(beg_factor * width / 2.0 / height);
270 		int end = (int) Math.floor(end_factor * width / 2.0 / height);
271 
272 		setFacetRow(facet,
273 					rot,
274 					beg,
275 					end,
276 					facet < 3 ? y : y + height/2,
277 					pixels);
278 
279 		return end - beg;
280 	}
